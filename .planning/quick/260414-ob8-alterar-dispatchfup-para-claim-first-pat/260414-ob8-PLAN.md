**Pattern to follow** (similar to runs dispatch, but with status revert):
```typescript
// 1. Claim first
const claimResult = await db.collection('fup').findOneAndUpdate(
  { _id: fupId, status: 'on' },
  { $set: { status: 'queued' } },
);
if (!claimResult) {
  this.logger.warn(`FUP ${String(fupId)} already claimed by another cycle`);
  return false;
}

// 2. POST to webhook
const success = await this.post(webhookUrl, fup);

// 3. Handle failure with revert
if (!success) {
  // Revert status back to 'on'
  await db.collection('fup').findOneAndUpdate(
    { _id: fupId, status: 'queued' },
    { $set: { status: 'on' } },
  );

  // 60s retry with same claim-first logic
  const retryFn = (): void => {
    void (async () => {
      // Retry also uses claim-first
      const retryClaimResult = await db.collection('fup').findOneAndUpdate(
        { _id: fupId, status: 'on' },
        { $set: { status: 'queued' } },
      );
      if (!retryClaimResult) return;

      const retrySuccess = await this.post(webhookUrl, fup);
      if (!retrySuccess) {
        // Revert retry claim
        await db.collection('fup').findOneAndUpdate(
          { _id: fupId, status: 'queued' },
          { $set: { status: 'on' } },
        );
      }
    })();
  };
  setTimeout(retryFn, 60_000);
  return false;
}

// 4. Success - status stays 'queued'
return true;
```

**Key difference from runs**: Runs use "POST-first then claim" because once queued they stay queued. FUPs need "claim-first then revert-on-failure" because failed dispatches should return to 'on' state for retry.
