**GREEN phase — implement dispatchMessage() in webhook-dispatch.service.ts:**

Add method after `dispatchFup()`, following the EXACT same structure. Key differences from dispatchFup:
- collection name: `'messages'`
- filter: `{ _id: messageId, messageStatus: 'pending' }`
- update: `{ $set: { messageStatus: 'processing' } }`
- warn message: `Message ${String(messageId)} already claimed by another cycle`
- parameter name: `message` (not `fup`)

```typescript
async dispatchMessage(db: Db, message: Document, webhookUrl: string): Promise<void> {
  const messageId = message['_id'] as ObjectId;
  const success = await this.post(webhookUrl, message);

  if (success) {
    const result = await db
      .collection('messages')
      .findOneAndUpdate(
        { _id: messageId, messageStatus: 'pending' },
        { $set: { messageStatus: 'processing' } },
      );
    if (!result) {
      this.logger.warn(
        `Message ${String(messageId)} already claimed by another cycle`,
      );
    }
    return;
  }

  // MSG-07: single non-blocking retry after 60s
  const retryFn = (): void => {
    void this.post(webhookUrl, message).then(async (retrySuccess) => {
      if (retrySuccess) {
        await db
          .collection('messages')
          .findOneAndUpdate(
            { _id: messageId, messageStatus: 'pending' },
            { $set: { messageStatus: 'processing' } },
          );
      }
      // MSG-08: if retry fails, leave message as 'pending' — next cycle picks up
    });
  };
  setTimeout(retryFn, 60_000);
}
```

```typescript
private async processDatabase(dbName: string): Promise<void> {
  const db: Db = this.mongoService.db(dbName);

  // --- timeTrigger-gated: runs + FUP ---
  const vars = await db.collection('vars').findOne<VarsDoc>({});
  if (!vars?.timeTrigger) {
    this.logger.warn(`[${dbName}] timeTrigger not found in vars — skipping`);
  } else if (!vars.timeTrigger.enabled) {
    this.logger.warn(`[${dbName}] timeTrigger.enabled is false — skipping`);
  } else if (!this.isWithinTimeWindow(vars.timeTrigger.morningLimit, vars.timeTrigger.nightLimit)) {
    // silent skip
  } else if (!this.isAllowedDay(vars.timeTrigger.allowedDays)) {
    // silent skip
  } else {
    // DETECT-03: fresh webhooks read every cycle
    const webhookDoc = await db.collection('webhooks').findOne<WebhookDoc>({});
    const webhookUrl = webhookDoc?.['Processador de Runs'];
    if (!webhookUrl) {
      this.logger.warn(`[${dbName}] "Processador de Runs" URL missing from webhooks — skipping`);
    } else {
      const runs: Document[] = await db.collection('runs').find({ runStatus: 'waiting', waitUntil: { $lte: Date.now() } }).toArray();
      for (const run of runs) {
        await this.webhookDispatchService.dispatch(db, run, webhookUrl);
      }
    }

    const fupWebhookUrl = webhookDoc?.['Gerenciador follow up'] as string | undefined;
    if (!fupWebhookUrl) {
      this.logger.warn(`[${dbName}] "Gerenciador follow up" URL missing from webhooks — skipping FUP dispatch`);
    } else {
      const fups: Document[] = await db.collection('fup').find({ status: 'on', nextInteractionTimestamp: { $lte: Date.now() } }).toArray();
      for (const fup of fups) {
        await this.webhookDispatchService.dispatchFup(db, fup, fupWebhookUrl);
      }
    }
  }

  // --- MSG-01/MSG-02/MSG-03: messages — NO time gate, NO day gate ---
  const messagesWebhookDoc = await db.collection('webhooks').findOne<WebhookDoc>({});
  const messagesWebhookUrl = messagesWebhookDoc?.['mensagens pendentes'] as string | undefined;
  if (!messagesWebhookUrl) {
    this.logger.warn(`[${dbName}] "mensagens pendentes" URL missing from webhooks — skipping messages dispatch`);
  } else {
    const messages: Document[] = await db.collection('messages').find({ messageStatus: 'pending' }).toArray();
    for (const message of messages) {
      await this.webhookDispatchService.dispatchMessage(db, message, messagesWebhookUrl);
    }
  }
}
```

NOTE: The webhooks collection is read twice (once inside the timeTrigger block, once for messages). This is intentional and acceptable — keeps the logic simple and aligns with the "fresh read every cycle" pattern (DETECT-03).
