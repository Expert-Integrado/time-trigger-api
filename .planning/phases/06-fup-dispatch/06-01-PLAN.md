      if (success) {

          .collection('fup')
          .findOneAndUpdate(

            { $set: { status: 'queued' } },

          );
        }
        return;

      const retryFn = (): void => {

          if (retrySuccess) {
            await db
              .collection('fup')
              .findOneAndUpdate(
                { _id: fupId, status: 'on' },
                { $set: { status: 'queued' } },
              );
          }
          // FUP-08: if retry fails, leave fup as 'on' — next cycle picks up
        });
      };
      setTimeout(retryFn, 60_000);
    }
